\begin{document}
\begin{tikzpicture}[

    block/.style = {draw, rectangle, minimum height=0.9cm, minimum width=1.3cm, align=center, thick},

% =========================================================

\node[block, below=1.5cm of plant] (uav) {UAV\\(Physical)};
\node[block, left=2.5cm of uav] (gyro) {Gyro};

% Gyro to Summing Junction
